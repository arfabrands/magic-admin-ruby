def logout_by_issuer(issuer)
  headers = MagicAdmin::Util.headers(magic.secret_key)
  options = { params: { issuer: issuer }, headers: headers }
  magic.http_client
       .call(:post, "/v2/admin/auth/user/logout", options)
end
